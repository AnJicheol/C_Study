public class Main {

public static void main(String[] args) {

People winter_people = new People();

winter_people.pants_change(new W_Pants());

winter_people.outer_change(new W_Outer());

winter_people.pants_print();

winter_people.outer_print();

  

People summer_people = new People();

summer_people.pants_change(new S_Pants());

summer_people.outer_change(new S_Outer());

summer_people.pants_print();

summer_people.outer_print();

}

}

  

class Pants_Clothes{

public void patns_put_on(){

}

}

class Outer_Clothes{

public void outer_put_on(){

}

}

  

class People{

Pants_Clothes pants_clothes;

Outer_Clothes outer_clothes;

public void pants_change(Pants_Clothes pc){

pants_clothes = pc;

}

public void outer_change(Outer_Clothes oc){

outer_clothes = oc;

}

public void pants_print(){

pants_clothes.patns_put_on();

}

public void outer_print(){

outer_clothes.outer_put_on();

}

}

  
  

class W_Pants extends Pants_Clothes{

public void patns_put_on(){

System.out.println("긴바지를 입었습니다.");

}

}

  

class S_Pants extends Pants_Clothes{

public void patns_put_on(){

System.out.println(" 반바지를 입었습니다.");

}

}

  

class W_Outer extends Outer_Clothes{

public void outer_put_on(){

System.out.println("패딩을 입었습니다.");

}

}

  

class S_Outer extends Outer_Clothes{

public void outer_put_on(){

System.out.println("반팔을 입었습니다.");

}

}
